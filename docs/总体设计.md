### 2.2 Agent 统一接口

```python
from abc import ABC, abstractmethod
from typing import AsyncGenerator
from dataclasses import dataclass

@dataclass
class AgentConfig:
    provider: str           # "openai" | "anthropic" | "deepseek" | "local"
    model: str              # "gpt-4o" | "claude-sonnet-4-20250514" | ...
    temperature: float      # 0.0 - 1.0
    max_tokens: int         # 最大输出 token
    system_prompt: str      # 自定义系统提示词
    api_key: str            # API Key

@dataclass
class AgentEvent:
    type: str               # "thinking" | "coding" | "submitted" | "error"
    content: str            # 内容
    timestamp: float        # 时间戳

@dataclass
class JudgeResult:
    status: str             # "accepted" | "wrong_answer" | ...
    passed_tests: int
    total_tests: int
    score: float
    comments: str

@dataclass
class DebateArgument:
    accepts_result: bool    # 是否接受评判结果
    argument: str           # 辩驳理由
    evidence: str           # 证据（如反例）

class BaseAgent(ABC):
    def __init__(self, config: AgentConfig):
        self.config = config
    
    @abstractmethod
    async def think(self, problem: str) -> AsyncGenerator[AgentEvent, None]:
        """思考阶段：分析题目，规划解题思路"""
        pass
    
    @abstractmethod
    async def code(self, problem: str, thinking: str) -> AsyncGenerator[AgentEvent, None]:
        """编码阶段：根据思考结果生成代码"""
        pass
    
    @abstractmethod
    async def debate(self, judge_result: JudgeResult) -> DebateArgument:
        """辩驳阶段：对评判结果提出异议"""
        pass
```

### 2.3 Agent 适配器实现

每个 provider 的适配器负责：
1. 将统一的 prompt 转换为 provider 特定的 API 格式
2. 处理 streaming 响应，转换为统一的 `AgentEvent`
3. 处理错误和重试逻辑

```python
# 示例：OpenAI Agent
class OpenAIAgent(BaseAgent):
    async def think(self, problem: str) -> AsyncGenerator[AgentEvent, None]:
        messages = [
            {"role": "system", "content": self.config.system_prompt},
            {"role": "user", "content": f"分析以下算法题目，给出解题思路：\n\n{problem}"}
        ]
        
        async for chunk in self.client.chat.completions.create(
            model=self.config.model,
            messages=messages,
            stream=True
        ):
            if chunk.choices[0].delta.content:
                yield AgentEvent(
                    type="thinking",
                    content=chunk.choices[0].delta.content,
                    timestamp=time.time()
                )
```

### 2.4 速度限制器（人机公平模式）

```python
class ThrottleController:
    """
    限制 Agent 的输出速度，模拟人类行为。
    用于人机对战模式，确保公平性。
    """
    
    def __init__(
        self,
        chars_per_second: float = 5.0,      # 每秒输出字符数
        think_delay_range: tuple = (2.0, 8.0),  # 思考延迟范围（秒）
        typo_probability: float = 0.02,     # 模拟打字错误概率
        pause_probability: float = 0.05     # 模拟停顿概率
    ):
        self.cps = chars_per_second
        self.think_delay_range = think_delay_range
        self.typo_prob = typo_probability
        self.pause_prob = pause_probability
    
    async def throttle_stream(
        self, stream: AsyncGenerator[AgentEvent, None]
    ) -> AsyncGenerator[AgentEvent, None]:
        """对 Agent 输出流施加速度限制"""
        
        # 模拟初始思考延迟
        delay = random.uniform(*self.think_delay_range)
        await asyncio.sleep(delay)
        
        async for event in stream:
            if event.type in ("thinking", "coding"):
                # 逐字符输出，模拟打字
                for char in event.content:
                    char_delay = 1.0 / self.cps
                    # 随机停顿
                    if random.random() < self.pause_prob:
                        await asyncio.sleep(random.uniform(0.5, 2.0))
                    else:
                        await asyncio.sleep(char_delay)
                    yield AgentEvent(type=event.type, content=char, timestamp=time.time())
            else:
                yield event
```

### 3.1 评判流程

```
Agent 提交代码
       │
       ▼

┌──────────────┐
```

```python
class CodeRunner:
    """代码执行器，对接 Judge0 API"""
    
    LANGUAGE_MAP = {
        "python": 71,      # Python 3
        "javascript": 63,  # JavaScript (Node.js)
        "cpp": 54,         # C++ (GCC)

        self,
        code: str,
        language: str,
        stdin: str,

        memory_limit: int = 262144

        resp = await self.client.post(
            f"{self.base_url}/submissions?wait=true",
            json=payload
        )
        result = resp.json()
        
        return ExecutionResult(
            stdout=result.get("stdout", ""),
            stderr=result.get("stderr", ""),
            status=result["status"]["description"],
            time=result.get("time"),
            memory=result.get("memory"),
```

### 4.2 辩驳数据模型

```python
@dataclass
class DebateRound:
    round_number: int
    agent_argument: str         # Agent 的辩驳论据
    evidence: str               # 支撑证据（如反例、执行结果）
    judge_response: str         # Judge 的回应
    verdict: str                # "accepted" | "rejected" | "partial"
    score_adjustment: float     # 分数调整值

class DebateManager:
    MAX_ROUNDS = 2              # 最多辩驳轮次
    ARBITRATION_THRESHOLD = 2   # 达到该轮次触发仲裁
    
    async def run_debate(
        self, agent: BaseAgent, judge: JudgeAgent, result: JudgeResult
    ) -> DebateOutcome:
        """执行辩驳流程"""
        rounds = []
        
        for round_num in range(self.MAX_ROUNDS):
            # Agent 提出辩驳
            argument = await agent.debate(result)
            
            if argument.accepts_result:
                return DebateOutcome(accepted=True, rounds=rounds)
            
            # Judge 回应辩驳
            response = await judge.respond_to_debate(
                original_result=result,
                argument=argument,
                previous_rounds=rounds
            )
            
            round_record = DebateRound(
                round_number=round_num + 1,
                agent_argument=argument.argument,
                evidence=argument.evidence,
                judge_response=response.reasoning,
                verdict=response.verdict,
                score_adjustment=response.score_adjustment
            )
            rounds.append(round_record)
            
            if response.verdict == "accepted":
                result.score += response.score_adjustment
                return DebateOutcome(accepted=True, rounds=rounds, new_score=result.score)
            
            if response.verdict == "partial":
                result.score += response.score_adjustment
        
        # 如果所有轮次辩驳未达成一致，触发仲裁
        if rounds[-1].verdict == "rejected":
            arbitration = await self.arbitrate(agent, judge, result, rounds)
            return arbitration
        
        return DebateOutcome(accepted=False, rounds=rounds, new_score=result.score)
    
    async def arbitrate(self, agent, judge, result, rounds) -> DebateOutcome:
        """仲裁：引入第三方 Agent 做最终裁决"""
        arbitrator = AgentFactory.create(self.arbitrator_config)
        
        context = self._format_debate_history(rounds)
        final_verdict = await arbitrator.arbitrate(
            original_result=result,
            debate_history=context
        )
        
        return DebateOutcome(
            accepted=True,
            rounds=rounds,
            arbitrated=True,
            new_score=final_verdict.final_score
        )
```

### 4.3 辩驳触发条件

Agent 可以在以下情况下发起辩驳：
1. **认为测试用例有误** — 提供反例证明自己代码的正确性
2. **认为评分不公平** — 指出代码质量被低估的原因
3. **认为超时判定不合理** — 说明算法复杂度和测试数据规模的关系
4. **认为部分通过应得更高分** — 论证接近正确解的价值

```python
from enum import Enum

class ContestState(Enum):
    WAITING = "waiting"       # 等待开始
    RUNNING = "running"       # Agent 正在解题
    JUDGING = "judging"       # 评判中
    DEBATING = "debating"     # 辩驳中
    FINISHED = "finished"     # 已结束

class ContestStateMachine:
    TRANSITIONS = {
        ContestState.WAITING: [ContestState.RUNNING],
        ContestState.RUNNING: [ContestState.JUDGING],
        ContestState.JUDGING: [ContestState.DEBATING, ContestState.FINISHED],
        ContestState.DEBATING: [ContestState.JUDGING, ContestState.FINISHED],
        ContestState.FINISHED: [],
    }
    
    def __init__(self):
        self.state = ContestState.WAITING
        self.history = []
    
    def can_transition(self, to_state: ContestState) -> bool:
        return to_state in self.TRANSITIONS[self.state]
    
    async def transition(self, to_state: ContestState):
        if not self.can_transition(to_state):
            raise InvalidTransitionError(
                f"Cannot transition from {self.state} to {to_state}"
            )
        self.history.append((self.state, to_state, time.time()))
        self.state = to_state
        await self.notify_listeners(to_state)
```

### 5.2 比赛管理器

```python
class ContestManager:
    """比赛全流程管理"""
    
    async def create_contest(self, config: ContestConfig) -> Contest:
        """创建比赛"""
        contest = Contest(
            id=uuid4(),
            title=config.title,
            problem_id=config.problem_id,
            participants=config.participants,
            mode=config.mode,
            state_machine=ContestStateMachine()
        )
        await self.db.save(contest)
        return contest
    
    async def start_contest(self, contest_id: str):
        """启动比赛"""
        contest = await self.db.get(contest_id)
        await contest.state_machine.transition(ContestState.RUNNING)
        
        # 分发题目给所有 Agent
        problem = await self.problem_repo.get(contest.problem_id)
        
        # 并发启动所有 Agent
        tasks = []
        for participant in contest.participants:
            agent = self.agent_factory.create(participant.agent_config)
            task = asyncio.create_task(
                self._run_agent(contest, agent, problem)
            )
            tasks.append(task)
        
        # 等待所有 Agent 完成或超时
        await asyncio.gather(*tasks)
        
        # 进入评判阶段
        await contest.state_machine.transition(ContestState.JUDGING)
        await self._judge_submissions(contest)
    
    async def _run_agent(self, contest, agent, problem):
        """运行单个 Agent 的解题流程"""
        # 思考阶段
        thinking_log = ""
        async for event in agent.think(problem.description):
            thinking_log += event.content
            await self.ws_hub.broadcast(contest.id, event)
        
        # 编码阶段
        code = ""
        async for event in agent.code(problem.description, thinking_log):
            code += event.content
            await self.ws_hub.broadcast(contest.id, event)
        
        # 保存提交
        submission = Submission(
            contest_id=contest.id,
            agent_id=agent.id,
            code=code,
            thinking_log=thinking_log,
            submitted_at=time.time()
        )
        await self.db.save(submission)
```

### 5.3 WebSocket 事件系统

```python
class WebSocketHub:
    """WebSocket 连接管理和事件广播"""
    
    def __init__(self):
        self.connections: dict[str, list[WebSocket]] = {}  # contest_id -> connections
    
    async def connect(self, contest_id: str, websocket: WebSocket):
        await websocket.accept()
        if contest_id not in self.connections:
            self.connections[contest_id] = []
        self.connections[contest_id].append(websocket)
    
    async def disconnect(self, contest_id: str, websocket: WebSocket):
        self.connections[contest_id].remove(websocket)
    
    async def broadcast(self, contest_id: str, event: dict):
        """向比赛的所有观众广播事件"""
        if contest_id in self.connections:
            for connection in self.connections[contest_id]:
                try:
                    await connection.send_json(event)
                except WebSocketDisconnect:
                    await self.disconnect(contest_id, connection)
```

### 7.2 完整表结构

```sql
-- 比赛表
CREATE TABLE contests (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title VARCHAR(200) NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'waiting',
    -- status: waiting | running | judging | debating | finished
    mode VARCHAR(20) NOT NULL DEFAULT 'agent_vs_agent',
    -- mode: agent_vs_agent | human_vs_agent
    problem_id UUID NOT NULL REFERENCES problems(id),
    config JSONB NOT NULL DEFAULT '{}',
    -- config 示例: {"time_limit": 300, "max_debate_rounds": 2}
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    started_at TIMESTAMP WITH TIME ZONE,
    finished_at TIMESTAMP WITH TIME ZONE
);

-- 题目表
CREATE TABLE problems (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title VARCHAR(200) NOT NULL,
    description TEXT NOT NULL,
    difficulty VARCHAR(10) NOT NULL DEFAULT 'medium',
    -- difficulty: easy | medium | hard
    time_limit_ms INT NOT NULL DEFAULT 2000,
    memory_limit_kb INT NOT NULL DEFAULT 262144,
    tags TEXT[] DEFAULT '{}',
    source VARCHAR(100),
    source_id VARCHAR(50),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 测试用例表
CREATE TABLE test_cases (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    problem_id UUID NOT NULL REFERENCES problems(id) ON DELETE CASCADE,
    input TEXT NOT NULL,
    expected_output TEXT NOT NULL,
    is_sample BOOLEAN DEFAULT FALSE,
    weight FLOAT DEFAULT 1.0,
    order_index INT NOT NULL DEFAULT 0
);

-- Agent 配置表
CREATE TABLE agent_configs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    provider VARCHAR(20) NOT NULL,
    -- provider: openai | anthropic | deepseek | google | local
    model VARCHAR(100) NOT NULL,
    parameters JSONB DEFAULT '{}',
    -- parameters 示例: {"temperature": 0.7, "max_tokens": 4096}
    system_prompt TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 比赛参赛者表
CREATE TABLE contest_participants (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    contest_id UUID NOT NULL REFERENCES contests(id) ON DELETE CASCADE,
    agent_config_id UUID REFERENCES agent_configs(id),
    participant_type VARCHAR(10) NOT NULL DEFAULT 'agent',
    -- participant_type: agent | human
    human_user_id UUID,
    throttle_config JSONB DEFAULT '{}',
    -- throttle_config 示例: {"chars_per_second": 5, "think_delay": 3}
    UNIQUE(contest_id, agent_config_id)
);

-- 提交表
CREATE TABLE submissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    contest_id UUID NOT NULL REFERENCES contests(id),
    participant_id UUID NOT NULL REFERENCES contest_participants(id),
    code TEXT NOT NULL,
    language VARCHAR(20) NOT NULL DEFAULT 'python',
    thinking_log TEXT,
    submitted_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 评判结果表
CREATE TABLE judge_results (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    submission_id UUID NOT NULL REFERENCES submissions(id) ON DELETE CASCADE,
    status VARCHAR(30) NOT NULL,
    -- status: accepted | wrong_answer | time_limit_exceeded | 
    --         runtime_error | compile_error
    passed_tests INT NOT NULL DEFAULT 0,
    total_tests INT NOT NULL DEFAULT 0,
    execution_time_ms INT,
    memory_used_kb INT,
    correctness_score FLOAT DEFAULT 0,
    efficiency_score FLOAT DEFAULT 0,
    quality_score FLOAT DEFAULT 0,
    total_score FLOAT DEFAULT 0,
    judge_comments TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 辩驳记录表
CREATE TABLE debate_rounds (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    judge_result_id UUID NOT NULL REFERENCES judge_results(id) ON DELETE CASCADE,
    round_number INT NOT NULL,
    agent_argument TEXT NOT NULL,
    agent_evidence TEXT,
    judge_response TEXT NOT NULL,
    verdict VARCHAR(10) NOT NULL,
    -- verdict: accepted | rejected | partial
    score_adjustment FLOAT DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    UNIQUE(judge_result_id, round_number)
);

-- 索引
CREATE INDEX idx_contests_status ON contests(status);
CREATE INDEX idx_submissions_contest ON submissions(contest_id);
CREATE INDEX idx_judge_results_submission ON judge_results(submission_id);
CREATE INDEX idx_test_cases_problem ON test_cases(problem_id);
CREATE INDEX idx_debate_rounds_result ON debate_rounds(judge_result_id);
```

## 九、安全设计

### 9.1 代码执行安全

- **网络隔离**：Judge0 容器禁用网络访问
- **资源限制**：CPU 时间、内存、进程数严格限制
- **文件系统**：只读挂载，禁止写入宿主机
- **超时保护**：硬超时自动 kill 进程

### 9.2 API 安全

- **API Key 加密存储**：不明文保存 LLM API Key
- **请求限流**：防止 API 滥用
- **输入验证**：所有用户输入严格校验
- **CORS 配置**：限制允许的域名

### 9.3 数据安全

- **敏感信息脱敏**：日志中不记录 API Key
- **数据库备份**：定期自动备份
- **访问控制**：基于角色的权限管理（后续扩展）
